### Hybrid Base-10/13 Conversion

```python
def base10_to_hybrid(n):
    """Convert Base-10 integer to hybrid Base-10/13"""
    if n == 0:
        return '0'
    
    digits = []
    symbols = '0123456789ϴλΔ'
    
    while n > 0:
        digits.append(symbols[n % 13])
        n //= 13
    
    return ''.join(reversed(digits))

def hybrid_to_base10(s):
    """Convert hybrid Base-10/13 string to Base-10 integer"""
    symbols = '0123456789ϴλΔ'
    result = 0
    
    for char in s:
        result = result * 13 + symbols.index(char)
    
    return result
```
